# ── Audio / Video ──────────────────────────────────────────────────────────────

# ── Transcription — faster-whisper (CTranslate2) ──────────────────────────────
faster-whisper>=1.0.0

# ── Translation — EuroLLM via HuggingFace transformers ────────────────────────
# EuroLLM-9B-Instruct is downloaded automatically on first run (~18 GB).

# ── Speaker Denoising (in priority order — first available is used) ────────────
# DeepFilterNet: best quality (neural, 48 kHz). Install without version pin;
# if it fails on your platform, noisereduce or FFmpeg anlmdn will be used instead.
deepfilternet
# noisereduce: spectral gating, pure Python, reliable fallback.
noisereduce>=3.0.0
# FFmpeg anlmdn is also tried as a final fallback (requires only ffmpeg system pkg).

# ── TTS — VoxCPM2 (primary) ───────────────────────────────────────────────────
# OpenBMB, Apache 2.0 license.  2B diffusion AR model, 48 kHz, 30 languages.
# Ultimate Cloning mode uses both reference audio + transcript for max fidelity.
voxcpm

# ── TTS — Coqui XTTS v2 (fallback if voxcpm not installed) ───────────────────
# Community-maintained PyPI fork of the original TTS package.
# Installs to same TTS.* namespace; no code changes needed.
coqui-tts
# Pin transformers so coqui-tts's dep resolution doesn't upgrade past 4.40.2.
# 02_pipeline.py also patches the two removed symbols at runtime as a belt-and-
# suspenders fix (isin_mps_friendly + is_torch_greater_or_equal).
# Install order matters: coqui-tts first, then force-reinstall transformers.

# ── SRT Alignment — WhisperX ──────────────────────────────────────────────────
